```javascript
// Admin: Service management
router.get('/mgmt/services', (req, res) => {
  const services = servicesLoader.getServices();
  res.render('admin/services', { services });
});

router.post('/mgmt/services', (req, res) => {
  const { action, serviceId, name, description, enabled, order } = req.body;
  
  if (action === 'toggle') {
    servicesLoader.updateService(serviceId, { enabled: enabled === 'true' });
  } else if (action === 'update') {
    servicesLoader.updateService(serviceId, { name, description });
  } else if (action === 'reorder') {
    const services = servicesLoader.getServices();
    const ids = services.map(s => s.id);
    const currentIdx = ids.indexOf(serviceId);
    if (currentIdx > 0) {
      [ids[currentIdx], ids[currentIdx - 1]] = [ids[currentIdx - 1], ids[currentIdx]];
      servicesLoader.reorderServices(ids);
    }
  } else if (action === 'reorderDown') {
    const services = servicesLoader.getServices();
    const ids = services.map(s => s.id);
    const currentIdx = ids.indexOf(serviceId);
    if (currentIdx < ids.length - 1) {
      [ids[currentIdx], ids[currentIdx + 1]] = [ids[currentIdx + 1], ids[currentIdx]];
      servicesLoader.reorderServices(ids);
    }
  }
  
  res.redirect('/mgmt/services');
});
```
