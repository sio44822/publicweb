Create `utils/services-loader.js` following the same pattern as statistics.js (fs-based data persistence):

```javascript
const fs = require('fs');
const path = require('path');

const DATA_DIR = path.join(__dirname, '..', 'data');
const SERVICES_FILE = path.join(DATA_DIR, 'services.json');

function ensureDataDir() {
  if (!fs.existsSync(DATA_DIR)) {
    fs.mkdirSync(DATA_DIR, { recursive: true });
  }
}

function getServices() {
  ensureDataDir();
  if (!fs.existsSync(SERVICES_FILE)) {
    return [];
  }
  const data = fs.readFileSync(SERVICES_FILE, 'utf8');
  const parsed = JSON.parse(data);
  return parsed.services || [];
}

function saveServices(services) {
  ensureDataDir();
  const data = { services };
  fs.writeFileSync(SERVICES_FILE, JSON.stringify(data, null, 2));
}

function getEnabledServices() {
  return getServices()
    .filter(s => s.enabled)
    .sort((a, b) => a.order - b.order);
}

function getServiceById(id) {
  return getServices().find(s => s.id === id);
}

function updateService(id, updates) {
  const services = getServices();
  const idx = services.findIndex(s => s.id === id);
  if (idx === -1) return false;
  services[idx] = { ...services[idx], ...updates };
  saveServices(services);
  return true;
}

function reorderServices(orderedIds) {
  const services = getServices();
  orderedIds.forEach((id, index) => {
    const service = services.find(s => s.id === id);
    if (service) service.order = index + 1;
  });
  saveServices(services);
}

module.exports = {
  getServices,
  getEnabledServices,
  getServiceById,
  updateService,
  reorderServices
};
```

Exports match the needs of both pages:
- getServices() — all services for admin page
- getEnabledServices() — filtered and sorted for user page
- getServiceById() — single service lookup
- updateService() — toggle enabled state (per D-04: enable/disable toggles)
- reorderServices() — update display order (per D-09: edit display order)
